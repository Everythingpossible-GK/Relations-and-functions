    <script>
        let currentTab = 'cartesian';
        let chart = null;
        let relState = { A: [1,2,3,4], B: [5,6,7,8], pairs: [] };
        let fnState = { A: [1,2,3,4], B: ['P','Q','R','S'], mapping: {} };

        function switchTab(id) {
            document.querySelectorAll('.content-section').forEach(s => s.classList.add('hidden'));
            document.querySelectorAll('.tab-btn').forEach(b => {
                b.classList.remove('tab-btn-active');
                b.classList.add('tab-btn-inactive');
            });
            document.getElementById(`section-${id}`).classList.remove('hidden');
            document.getElementById(`tab-${id}`).classList.add('tab-btn-active');
            
            if(id === 'relations') initRelations();
            if(id === 'functions') initFunctions();
            if(id === 'gallery' && !chart) plotFn('identity');
            if(id === 'algebra') calculateAlgebra();
        }

        // --- Cartesian ---
        function updateCartesian() {
            const a = document.getElementById('setA').value.split(',').map(s => s.trim()).filter(s => s);
            const b = document.getElementById('setB').value.split(',').map(s => s.trim()).filter(s => s);
            const res = document.getElementById('cartesianResult');
            res.innerHTML = '';
            a.forEach(vA => b.forEach(vB => {
                const s = document.createElement('span');
                s.className = "px-2 py-1 bg-white border rounded shadow-sm text-indigo-600";
                s.textContent = `(${vA},${vB})`;
                res.appendChild(s);
            }));
            document.getElementById('productCount').innerText = `n(A×B) = ${a.length * b.length}`;
        }

        // --- Relations ---
        function initRelations() {
            const cA = document.getElementById('rel-setA');
            const cB = document.getElementById('rel-setB');
            cA.innerHTML = ''; cB.innerHTML = '';
            relState.A.forEach(v => {
                const b = document.createElement('button');
                b.className = "w-10 h-10 rounded-full border-2 border-indigo-400 bg-white font-bold";
                b.textContent = v;
                b.onclick = (e) => { pendingA = {val:v, el:e.target}; drawLines(); };
                cA.appendChild(b);
            });
            relState.B.forEach(v => {
                const b = document.createElement('button');
                b.className = "w-10 h-10 rounded-full border-2 border-emerald-400 bg-white font-bold";
                b.textContent = v;
                b.onclick = () => { if(pendingA) { relState.pairs.push({a:pendingA.val, b:v}); pendingA=null; drawLines(); updateRelSummary(); } };
                cB.appendChild(b);
            });
            drawLines();
        }

        let pendingA = null;
        function drawLines() {
            const canvas = document.getElementById('relCanvas');
            const ctx = canvas.getContext('2d');
            ctx.clearRect(0, 0, 300, 400);
            const rect = canvas.getBoundingClientRect();
            relState.pairs.forEach(p => {
                const bA = Array.from(document.querySelectorAll('#rel-setA button')).find(x => x.textContent == p.a);
                const bB = Array.from(document.querySelectorAll('#rel-setB button')).find(x => x.textContent == p.b);
                if(bA && bB) {
                    ctx.beginPath();
                    ctx.moveTo(0, bA.getBoundingClientRect().top - rect.top + 20);
                    ctx.lineTo(300, bB.getBoundingClientRect().top - rect.top + 20);
                    ctx.strokeStyle = '#6366f1'; ctx.stroke();
                }
            });
        }

        function updateRelSummary() {
            const d = [...new Set(relState.pairs.map(p => p.a))].sort();
            const r = [...new Set(relState.pairs.map(p => p.b))].sort();
            document.getElementById('domain-text').innerText = `{${d.join(',')}}`;
            document.getElementById('range-text').innerText = `{${r.join(',')}}`;
            document.getElementById('codomain-text').innerText = `{${relState.B.join(',')}}`;
        }

        function resetRelations() { relState.pairs = []; drawLines(); updateRelSummary(); }

        // --- Functions ---
        function initFunctions() {
            const ui = document.getElementById('fn-tester-ui'); ui.innerHTML = '';
            const cA = document.createElement('div'); cA.className = "flex flex-col gap-3";
            fnState.A.forEach(v => {
                const b = document.createElement('button');
                b.className = "w-12 h-12 border-2 rounded-lg font-bold hover:bg-indigo-50";
                b.id = `fna-${v}`; b.textContent = v;
                b.onclick = () => { selA = v; document.querySelectorAll('[id^=fna-]').forEach(el=>el.style.borderColor='#6366f1'); b.style.borderColor='#4338ca'; };
                cA.appendChild(b);
            });
            const cB = document.createElement('div'); cB.className = "flex flex-col gap-3";
            fnState.B.forEach(v => {
                const b = document.createElement('button');
                b.className = "w-12 h-12 border-2 rounded-lg font-bold hover:bg-gray-50";
                b.textContent = v;
                b.onclick = () => { if(selA) { fnState.mapping[selA] = v; updateFnUI(); } };
                cB.appendChild(b);
            });
            ui.appendChild(cA); ui.appendChild(cB);
        }

        let selA = null;
        function updateFnUI() {
            Object.entries(fnState.mapping).forEach(([k,v]) => {
                const el = document.getElementById(`fna-${k}`);
                el.classList.add('bg-indigo-600', 'text-white');
                el.innerHTML = `${k}<br><span class='text-[8px]'>→ ${v}</span>`;
            });
            const keys = Object.keys(fnState.mapping);
            const status = document.getElementById('fn-status');
            if(keys.length === fnState.A.length) {
                status.innerText = "✅ VALID FUNCTION";
                status.className = "inline-block px-8 py-4 rounded-2xl text-xl font-bold bg-green-50 text-green-700 border-green-200";
            } else {
                status.innerText = "NOT A FUNCTION YET (Mapping incomplete)";
                status.className = "inline-block px-8 py-4 rounded-2xl text-xl font-bold bg-amber-50 text-amber-600 border-amber-200";
            }
        }

        // --- Expanded Gallery ---
        function plotFn(type) {
            const ctx = document.getElementById('graphCanvas').getContext('2d');
            const tit = document.getElementById('graphTitle');
            const des = document.getElementById('graphDesc');
            if(chart) chart.destroy();
            let labels = [], data = [], stepped = false, radius = 2;

            switch(type) {
                case 'identity': tit.innerText = "Identity Function: f(x) = x"; des.innerText = "Domain: R, Range: R";
                    for(let x=-5; x<=5; x+=0.5) { labels.push(x); data.push(x); } break;
                case 'constant': tit.innerText = "Constant Function: f(x) = 3"; des.innerText = "Domain: R, Range: {3}";
                    for(let x=-5; x<=5; x+=0.5) { labels.push(x); data.push(3); } break;
                case 'modulus': tit.innerText = "Modulus Function: f(x) = |x|"; des.innerText = "Domain: R, Range: [0, ∞)";
                    for(let x=-5; x<=5; x+=0.5) { labels.push(x); data.push(Math.abs(x)); } break;
                case 'signum': tit.innerText = "Signum Function"; des.innerText = "Domain: R, Range: {-1, 0, 1}";
                    stepped = true; for(let x=-5; x<=5; x+=0.1) { labels.push(x.toFixed(1)); data.push(x>0?1:(x<0?-1:0)); } break;
                case 'gif': tit.innerText = "Greatest Integer Function: f(x) = [x]"; des.innerText = "Domain: R, Range: Z";
                    stepped = true; radius = 0; for(let x=-3; x<=3; x+=0.1) { labels.push(x.toFixed(1)); data.push(Math.floor(x)); } break;
                case 'square': tit.innerText = "Polynomial: f(x) = x²"; des.innerText = "Domain: R, Range: [0, ∞)";
                    for(let x=-5; x<=5; x+=0.2) { labels.push(x.toFixed(1)); data.push(x*x); } break;
                case 'cube': tit.innerText = "Polynomial: f(x) = x³"; des.innerText = "Domain: R, Range: R";
                    for(let x=-3; x<=3; x+=0.2) { labels.push(x.toFixed(1)); data.push(x*x*x); } break;
                case 'rational': tit.innerText = "Rational Function: f(x) = 1/x"; des.innerText = "Domain: R-{0}, Range: R-{0}";
                    for(let x=-4; x<=4; x+=0.2) { if(Math.abs(x) < 0.1) continue; labels.push(x.toFixed(1)); data.push(1/x); } break;
            }

            chart = new Chart(ctx, {
                type: 'line',
                data: { labels, datasets: [{ label:'f(x)', data, borderColor:'#4f46e5', stepped, pointRadius:radius, tension:0.1 }] },
                options: { responsive:true, maintainAspectRatio:false, scales: { y: { min:-5, max:5 } } }
            });
        }

        // --- Algebra of Functions ---
        function calculateAlgebra() {
            const x = parseFloat(document.getElementById('alg-x').value);
            const fx = x * x;
            const gx = 2 * x + 1;
            const res = document.getElementById('algebra-results');
            res.innerHTML = `
                <div class="flex justify-between border-b py-2"><span>$f(x) = x^2$</span><b>${fx}</b></div>
                <div class="flex justify-between border-b py-2"><span>$g(x) = 2x+1$</span><b>${gx}</b></div>
                <div class="flex justify-between py-2 text-indigo-700 font-bold"><span>$(f+g)(x)$</span><span>${fx + gx}</span></div>
                <div class="flex justify-between py-2 text-red-600 font-bold"><span>$(f-g)(x)$</span><span>${fx - gx}</span></div>
                <div class="flex justify-between py-2 text-emerald-700 font-bold"><span>$(fg)(x)$</span><span>${fx * gx}</span></div>
                <div class="flex justify-between py-2 text-purple-700 font-bold"><span>$(f/g)(x)$</span><span>${gx !== 0 ? (fx / gx).toFixed(2) : 'Undefined'}</span></div>
            `;
        }

        window.onload = () => { updateCartesian(); updateRelSummary(); };
    </script>
